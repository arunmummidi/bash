# Install UI tools
apt install openbox lxterminal xinit x11-server-utils

# Configure openbox window manager
# Add application entries in
/etc/xdg/openbox/menu.xml

# Launch windows maximized
install -Dv /dev/null  ~/.config/openbox/rc.yaml
echo "    <application class="*">
        <maximized>yes</maximized>
    </application>" >~/.config/openbox/rc.yaml

# Remove the configuration of unused
# interface from netplan
mv /etc/netplan/00-installer-config.yaml /etc/netplan/00-installer-config.yaml.original
netplan generate
netplan try

# Troubleshoot network
ip route show default # check if default route is good
ip route add default via 192.168.1.1 # add default route to gateway/ router if route is corrupt/missing

# Change driver level display backlight
/sys/class/backlight/intel_backlight/brightness

# Configure basic TCP/UDP security
apt install ufw && ufw allow from 192.168.0.0/27 to any port 22 # Making sure not getting locked out
ufw allow from 0.0.0.0 to any port <port> proto tcp # Exposes to internet!!
ufw show added
ufw enable

# Tune volume on console
amixer -D pulse sset Master 5%+
pulseaudio -k && pulseaudio -D && alsa force-reload

# Volume control GUI
pavucontrol
